                        'status': 'downloading', 

                        'downloaded_bytes': downloaded_data_len, 

                        'total_bytes_estimate': data_len, 

                        'tmpfilename': tmpfilename, 

                        'filename': filename, 

                        'eta': eta, 

                        'elapsed': time_now - start, 

                        'speed': speed, 

                    })

                else:

                            'filename': filename, 

                            'status': 'downloading', 

                            'elapsed': time_now - start, 

                            'speed': speed, 

                        })

            basic_args += ['--live']
